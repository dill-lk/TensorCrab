1/// Shape and row-major strides for an N-dimensional tensor.
2///
3/// Strides encode how many elements to skip in flat storage to advance by one
4/// step along each dimension, enabling zero-copy views (e.g. transpose).
5#[derive(Debug, Clone, PartialEq)]
6pub struct Shape {
7    /// Length of each dimension (e.g. `[3, 4]` for a 3 × 4 matrix).
8    pub dims: Vec<usize>,
9    /// Number of elements to skip per step along each dimension.
10    pub strides: Vec<usize>,
11}
12
13impl Shape {
14    /// Creates a row-major `Shape` from the given dimension sizes.
15    ///
16    /// For a shape `[d0, d1, ..., dn]` the strides are computed as:
17    /// `strides[i] = strides[i+1] * dims[i+1]`, with `strides[ndim-1] = 1`.
18    ///
19    /// # Example
20    /// ```
21    /// use tensor_crab::tensor::shape::Shape;
22    /// let s = Shape::row_major(&[3, 4]);
23    /// assert_eq!(s.strides, vec![4, 1]);
24    /// ```
25    pub fn row_major(dims: &[usize]) -> Self {
26        let ndim = dims.len();
27        let mut strides = vec![1usize; ndim];
28        for i in (0..ndim.saturating_sub(1)).rev() {
29            strides[i] = strides[i + 1] * dims[i + 1];
30        }
31        Self {
32            dims: dims.to_vec(),
33            strides,
34        }
35    }
36
37    /// Returns the total number of elements (product of all dimensions).
38    pub fn numel(&self) -> usize {
39        self.dims.iter().product()
40    }
41
42    /// Returns the number of dimensions.
43    pub fn ndim(&self) -> usize {
44        self.dims.len()
45    }
46
47    /// Computes the flat storage index from a multi-dimensional index array.
48    ///
49    /// `flat = sum(indices[i] * strides[i])` for all `i`.
50    pub fn flat_index(&self, indices: &[usize]) -> usize {
51        indices
52            .iter()
53            .zip(self.strides.iter())
54            .map(|(idx, stride)| idx * stride)
55            .sum()
56    }
57}
